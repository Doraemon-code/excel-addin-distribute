### 步骤 4：自动检测 Office 版本并写注册表

```python
def get_office_version() -> str:
    """自动检测已安装的 Office 版本，返回版本号字符串如 '16.0'"""
    for ver in ["16.0", "15.0", "14.0"]:
        try:
            winreg.OpenKey(
                winreg.HKEY_CURRENT_USER,
                f"Software\\Microsoft\\Office\\{ver}\\Excel\\Options"
            )
            return ver
        except FileNotFoundError:
            continue
    return "16.0"


def register_addin(xlam_path: str, log_callback):
    version = get_office_version()
    key_path = f"Software\\Microsoft\\Office\\{version}\\Excel\\Options"
    key = winreg.OpenKey(
        winreg.HKEY_CURRENT_USER, key_path, 0, winreg.KEY_ALL_ACCESS
    )
    # 查找可用的 OPEN 键名，避免覆盖其他加载项
    # 若已存在指向本 xlam 的键则复用
    i = 0
    target_key_name = None
    while True:
        name = "OPEN" if i == 0 else f"OPEN{i}"
        try:
            val, _ = winreg.QueryValueEx(key, name)
            if XLAM_FILENAME in val:
                target_key_name = name  # 复用已有键
                break
            i += 1
        except FileNotFoundError:
            target_key_name = name      # 找到空位
            break
    winreg.SetValueEx(
        key, target_key_name, 0, winreg.REG_SZ, f'"{xlam_path}"'
    )
    winreg.CloseKey(key)
    log_callback(f"📝 注册表已写入（Office {version}）：{target_key_name} = \"{xlam_path}\"")
```

## Git Hooks 设计（VBA 代码版本管理）

### 概述

VBA 代码存储在 xlam 二进制文件中，无法直接用 git 进行 diff 和版本追踪。通过 git hooks 在每次提交前自动导出 VBA 代码到文本文件，实现：

- ✅ 代码变更可追溯
- ✅ 支持代码 review
- ✅ 便于问题排查

### hooks 目录结构

```
hooks/
├── pre-commit           # 提交前自动导出 VBA 代码
├── hooks-config.json    # 配置文件
└── vba_exporter.py      # VBA 导出工具
```

### hooks-config.json

```json
{
  "xlam_path": "./MyAddin.xlam",
  "export_dir": "./vba-src",
  "subdirs": {
    "modules": "modules",
    "classes": "classes",
    "sheets": "sheets"
  },
  "export_format": "auto",
  "generate_manifest": true
}
```

### pre-commit 钩子流程

```
git commit
    ↓
触发 pre-commit 钩子
    ↓
检测暂存区是否有 .xlam 文件变更
    ↓
是 → 调用 vba_exporter.py 导出代码
    ↓
导出文件自动 git add
    ↓
提交完成（包含 xlam + VBA 源码）
```

### pre-commit 脚本

```bash
#!/bin/bash

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(dirname "$(dirname "$SCRIPT_DIR")")"
CONFIG_FILE="$SCRIPT_DIR/hooks-config.json"

echo "🔍 检测 VBA 代码变更..."

# 检查是否有 .xlam 文件被修改
CHANGED_XLAM=$(git diff --cached --name-only --diff-filter=M | grep '\.xlam$')

if [ -z "$CHANGED_XLAM" ]; then
    echo "✅ 无 xlam 文件变更，跳过 VBA 导出"
    exit 0
fi

# 读取配置
XLAM_PATH=$(jq -r '.xlam_path' "$CONFIG_FILE")
EXPORT_DIR=$(jq -r '.export_dir' "$CONFIG_FILE")

# 检查 Python 和导出工具
if ! command -v python &> /dev/null; then
    echo "❌ 错误：python 未安装"
    exit 1
fi

echo "📦 正在导出 VBA 代码..."

# 调用导出工具
python "$SCRIPT_DIR/vba_exporter.py" \
    --input "$REPO_ROOT/$XLAM_PATH" \
    --output "$REPO_ROOT/$EXPORT_DIR"

if [ $? -eq 0 ]; then
    echo "✅ VBA 代码导出成功"
    git add "$REPO_ROOT/$EXPORT_DIR"
    echo "📝 已添加导出文件到暂存区"
else
    echo "❌ VBA 代码导出失败"
    exit 1
fi
```

### vba_exporter.py 实现

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
VBA 代码导出工具
从 xlam 文件中导出 VBA 模块和 CustomUI XML
"""

import argparse
import json
import hashlib
import zipfile
from pathlib import Path
from datetime import datetime


def calc_hash(data: bytes) -> str:
    """计算 SHA256 哈希"""
    return hashlib.sha256(data).hexdigest()


def export_vba_code(xlam_path: str, output_dir: str):
    """从 xlam 文件导出 VBA 代码"""
    xlam_path = Path(xlam_path)
    output_dir = Path(output_dir)

    if not xlam_path.exists():
        print(f"❌ 文件不存在：{xlam_path}")
        return False

    # 创建输出目录
    (output_dir / "modules").mkdir(parents=True, exist_ok=True)
    (output_dir / "classes").mkdir(parents=True, exist_ok=True)
    (output_dir / "sheets").mkdir(parents=True, exist_ok=True)
    (output_dir / "customUI").mkdir(parents=True, exist_ok=True)

    modules_manifest = []

    try:
        with zipfile.ZipFile(xlam_path, 'r') as zip_ref:
            # 提取 vbaProject.bin（二进制，无法 diff）
            if 'xl/vbaProject.bin' in zip_ref.namelist():
                bin_data = zip_ref.read('xl/vbaProject.bin')
                with open(output_dir / "vbaProject.bin", 'wb') as f:
                    f.write(bin_data)
                modules_manifest.append({
                    "name": "vbaProject",
                    "type": "binary",
                    "file": "vbaProject.bin",
                    "hash": calc_hash(bin_data)
                })
                print(f"📦 导出：vbaProject.bin ({len(bin_data)} bytes)")

            # 提取 CustomUI XML
            for name in zip_ref.namelist():
                if name.startswith('customUI/'):
                    xml_data = zip_ref.read(name)
                    out_path = output_dir / name
                    out_path.parent.mkdir(parents=True, exist_ok=True)
                    with open(out_path, 'wb') as f:
                        f.write(xml_data)
                    modules_manifest.append({
                        "name": Path(name).stem,
                        "type": "customui",
                        "file": name,
                        "hash": calc_hash(xml_data)
                    })
                    print(f"📦 导出：{name}")

    except zipfile.BadZipFile:
        print("❌ 无效的 xlam 文件（不是有效的 ZIP 格式）")
        return False

    # 生成模块清单
    manifest = {
        "exported_at": datetime.now().isoformat(),
        "xlam_file": xlam_path.name,
        "xlam_hash": calc_hash(xlam_path.read_bytes()),
        "modules": modules_manifest
    }

    manifest_path = output_dir / "vba-modules.json"
    with open(manifest_path, 'w', encoding='utf-8') as f:
        json.dump(manifest, f, ensure_ascii=False, indent=2)
    print(f"📋 生成清单：{manifest_path}")

    return True


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="VBA 代码导出工具")
    parser.add_argument("--input", required=True, help="输入的 xlam 文件路径")
    parser.add_argument("--output", required=True, help="输出目录")
    args = parser.parse_args()

    success = export_vba_code(args.input, args.output)
    exit(0 if success else 1)
```
